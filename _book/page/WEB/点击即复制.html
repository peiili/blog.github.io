网页中点击复制

HTML点击复制

使用插件 cli

官网：https://github.com/zenorocha/clipboard.js.git

安装方法 ：npm install clipboard --save

npm install clipboard —save

在vue中的使用：

<template>

    <div id="app">
        <h1 style="font-size:2em;">网页中点击复制页面</h1>
        <div class="content1">
        //被复制内容
          <i id="msg">https://www.baidu.com2222</i>
          <button  class="copy" aria-hidden="true" data-clipboard-target="#msg" @click = "copy">点击复制</button>
        </div>
      <!-- <textarea name="" id="" cols="30" rows="10"></textarea> -->
    </div>

</template>

<script>

//引入 clipboard

import Clipboard from "clipboard"

export default {
  methods: {
     copy() {
      let clipboard = new Clipboard('.copy')
      //复制成功
      clipboard.on('success',e =>{
        alert('已复制到剪切板')
        clipboard.destroy()
      })
      //复制失败
      clipboard.on('error',e =>{
        alert('复制失败')
        clipboard.destroy()
      })
    }
  }
}
</script>

在传统HTML中使用

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta >
    <title>点击即复制</title>
    <script src="https://cdn.bootcss.com/clipboard.js/2.0.1/clipboard.js"></script>
</head>
<body>
    <input id="foo" value="https://github.com/zenorocha/clipboard.js.git">

<!-- Trigger -->
    <button class="btn" data-clipboard-target="#foo">
        <img src="assets/clippy.svg" alt="Copy to clipboard">
    </button>
    <textarea name="" id="" cols="30" rows="10"></textarea>
</body>
</html>

<script>
    var clipboard = new ClipboardJS('.btn');

    clipboard.on('success', function(e) {
        alert('已将'+e.text+'复制到剪切板')
        e.clearSelection();
    });

    clipboard.on('error', function(e) {
        console.error('Action:', e.action);
        console.error('Trigger:', e.trigger);
    });
</script>

插件中的事件:

var clipboard = new ClipboardJS('.btn');

clipboard.on('success', function(e) {
    console.info('Action:', e.action);
    console.info('Text:', e.text);
    console.info('Trigger:', e.trigger);

    e.clearSelection();
});

clipboard.on('error', function(e) {
    console.error('Action:', e.action);
    console.error('Trigger:', e.trigger);
});
